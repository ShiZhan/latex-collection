\usetikzlibrary{lindenmayersystems}

\begin{document}

\pgfdeclarelindenmayersystem{Fractal plant}{
  \rule{X -> F-[[X]+X]+F[+FX]-X}
  \rule{F -> FF}}

\begin{tikzpicture}

% draw edge
\draw[black!10] (0,0) circle (9.1cm);
\draw[black!10] (0,0) circle (0.5cm);

% draw dial
\def \n {12}
\def \radius {7.6cm}

\foreach \s in {1,...,\n}
{
  \node[font=\fontsize{70}{70}\selectfont] at ({360/\n * (2 - (\s - 1))}:\radius) {\s};
}

% draw decoration
\draw [green!50!black, rotate=90, shift={(-6 cm, 0 cm)}]
[l-system={Fractal plant, axiom=X, order=6, step=2pt, angle=25}]
lindenmayer system; 

\end{tikzpicture}
